Hi,
Mình có đọan code ở C#, nghiên cứu nhé.
Add reference: Microsoft.DirectX.AudioVideoPlayback
Kéo thả 1 Panel, 1 OpenFileDialog, 4 button (Open File, Play, Pause, Stop)
Khai báo thêm:
using Microsoft.DirectX.AudioVideoPlayback;

Code:
private Video _video = null;
private void MainForm_Load(object sender, EventArgs e)
{
   ControlEnable();
}
 
void ControlEnable()
{
   if (_video == null)
   {
       cmdPlay.Enabled = false;
       cmdStop.Enabled = false;
       cmdPause.Enabled = false;
   }
   else
   {
       cmdPlay.Enabled = true;
       cmdStop.Enabled = true;
       cmdPause.Enabled = true;
   }
}
 
private void cmdPlay_Click(object sender, EventArgs e)
{
   if (_video != null)
   {
       _video.Play();
   }
}
 
private void cmdPause_Click(object sender, EventArgs e)
{
   if (_video != null)
   {
       _video.Pause();
   }
}
 
private void cmdStop_Click(object sender, EventArgs e)
{
   if (_video != null)
   {
      _video.Stop();
   }
}
 
private void cmdOpenFile_Click(object sender, EventArgs e)
{
   OpenVideo();
}
 
void OpenVideo()
{
   openFileDialog1.InitialDirectory = Application.StartupPath;
   if (openFileDialog1.ShowDialog() == DialogResult.OK)
   {
      // open the video
      int height = videoPanel.Height;
      int width = videoPanel.Width;
      // dispose of the old video
      if (_video != null)
      {
          _video.Dispose();
      }
      // open a new video
      _video = newVideo(openFileDialog1.FileName);
      // assign the win form control that will contain the video
      _video.Owner = videoPanel;
      // resize to fit in the panel
      videoPanel.Width = width;
      videoPanel.Height = height;
      // play the first frame of the video so we can identify it
      _video.Play();
      _video.Pause();
   }
  // enable video buttons
   ControlEnable();
}
